\section*{Formulário de Apoio}

\begin{itemize}
    \item \textbf{Tensão de Pico e RMS:} $V_m = V_{rms} \cdot \sqrt{2}$
    \item \textbf{Retificador de Meia-Onda (Ideal):} $V_{dc} \approx 0,318 V_m$
    \item \textbf{Retificador de Onda Completa (Ideal):} $V_{dc} \approx 0,636 V_m$
    \item \textbf{Energia do Fóton / Gap:} $E_g = \frac{h \cdot c}{\lambda} \approx \frac{1240\text{ eV}\cdot\text{nm}}{\lambda}$
    \item \textbf{PIV (Ponte):} $PIV \geq V_m$
\end{itemize}

\section*{Gabarito Detalhado}

\begin{enumerate}[label=\textbf{Questão \arabic*:}, leftmargin=*]

    \item \textbf{Retificador com Bias (Fig. 1):} O diodo conduz apenas quando $v_i > 2\text{ V}$. Quando $v_i < 2\text{ V}$, o diodo está em corte e $v_d = v_i - 2\text{ V}$. O valor mínimo de $v_d$ (pico negativo) ocorre em $-(V_m + 2\text{ V})$.
    
    \item \textbf{Carga em Paralelo (Fig. 2):} Quando o diodo conduz ($v_i > 2\text{ V}$), $v_L = v_i$. A corrente $i_L$ segue o formato da tensão, com pico $I_{L,p} = V_{L,p} / 10\text{ k}\Omega$. Note que a fonte de $2\text{ V}$ atua como o nível de ceifamento.

    \item \textbf{Ponte 120V rms:} 
    \begin{itemize}
        \item $V_m = 120\sqrt{2} \approx 169,7\text{ V}$. 
        \item $V_{L,p} = 169,7 - 1,4 = 168,3\text{ V}$. 
        \item $V_{dc} = 0,636 \cdot 168,3 \approx \mathbf{107,04\text{ V}}$.
        \item $PIV = V_{L,p} + 0,7 = \mathbf{169,7\text{ V}}$.
    \end{itemize}

    \item \textbf{Análise da Figura 3:} $v_o$ é uma onda retificada completa com pico de $100\text{ V}$. $PIV = \mathbf{100\text{ V}}$. $I_{max} = 100\text{ V} / 2,2\text{ k}\Omega \approx \mathbf{45,45\text{ mA}}$.

    \item \textbf{Cálculo de Comprimento de Onda:} 
    $\lambda = 1240 / 0,67 \approx \mathbf{1850,7\text{ nm}}$. Como $\lambda > 700\text{ nm}$, o nível de energia é \textbf{inferior} ao visível (região do infravermelho).

    \item \textbf{Comparação de Materiais:} $V_K(Ge) \approx 0,3\text{V}$, $V_K(Si) \approx 0,7\text{V}$, $V_K(LED Blue) \approx 3,0\text{V}$. A tensão depende de $E_g$; fótons de maior energia (azul/UV) exigem maior queda de tensão para a recombinação.

    \item \textbf{Intensidade Angular:} $0,75$ é $25\%$ da intensidade máxima. Consultando curvas típicas de distribuição espacial, isso ocorre geralmente em ângulos próximos a $\mathbf{40^\circ \text{ a } 50^\circ}$. A redução $> 50\%$ ocorre para ângulos superiores ao ângulo de meia-intensidade (tipicamente $30^\circ$ em LEDs diretivos).

    \item \textbf{Eficiência e Corrente:} 
    \begin{itemize}
        \item \textbf{a) e b)} O aumento percentual é maior em correntes baixas ($5 \to 10\text{ mA}$) devido à inclinação da curva de eficiência. 
        \item \textbf{c)} A eficiência tende a saturar em correntes elevadas ($> 30\text{ mA}$), onde aumentos na corrente de pico geram ganhos mínimos de intensidade luminosa relativa.
    \end{itemize}

\end{enumerate}
